# DronePi Python requirements
# For CI and laptop development (no conda, no hardware).
# On the Pi use the dronepi conda environment instead — open3d, pdal, and
# pymeshlab have no aarch64 pip wheels and must come from conda-forge.
#
# Install: pip install -r requirements.txt

lark
colcon-common-extensions
colcon-ros
osrf-pycommon
rosdep

# ── Camera / image ────────────────────────────────────────────────────────────

                               # ortho_tools, texture_projector
# picamera2 is Pi-only — not installable on x86. Install on Pi:
#   pip install picamera2 --break-system-packages
